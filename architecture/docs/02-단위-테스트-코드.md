```typescript
// tests/auth/phone-request.test.ts
import request from 'supertest';
import app from '../../src/app';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

describe('POST /api/v1/auth/phone/request', () => {
  beforeEach(async () => {
    await prisma.setting.deleteMany();
  });

  it('유효한 전화번호로 인증 요청 성공', async () => {
    const response = await request(app)
      .post('/api/v1/auth/phone/request')
      .send({ phone: '010-1234-5678' });

    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
    expect(response.body.message).toBe('인증번호가 전송되었습니다');
    expect(response.body.expires_in).toBe(180);
  });

  it('전화번호 없이 요청 시 400 에러', async () => {
    const response = await request(app)
      .post('/api/v1/auth/phone/request')
      .send({});

    expect(response.status).toBe(400);
    expect(response.body.success).toBe(false);
    expect(response.body.error).toBe('전화번호가 필요합니다');
  });

  it('잘못된 전화번호 형식으로 요청 시 400 에러', async () => {
    const response = await request(app)
      .post('/api/v1/auth/phone/request')
      .send({ phone: '1234567890' });

    expect(response.status).toBe(400);
    expect(response.body.success).toBe(false);
  });

  it('DB에 인증번호가 저장되는지 확인', async () => {
    await request(app)
      .post('/api/v1/auth/phone/request')
      .send({ phone: '010-1234-5678' });

    const setting = await prisma.setting.findUnique({
      where: { key: 'verify_010-1234-5678' }
    });

    expect(setting).not.toBeNull();
    expect(setting?.type).toBe('auth');
    const value = JSON.parse(setting!.value);
    expect(value.code).toMatch(/^\d{6}$/);
  });
});
```

### 2.2 전화번호 인증 검증

```typescript
// tests/auth/phone-verify.test.ts
import request from 'supertest';
import app from '../../src/app';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

describe('POST /api/v1/auth/phone/verify', () => {
  let verifyCode: string;

  beforeEach(async () => {
    await prisma.user.deleteMany();
    await prisma.setting.deleteMany();

    const response = await request(app)
      .post('/api/v1/auth/phone/request')
      .send({ phone: '010-1234-5678' });

    const setting = await prisma.setting.findUnique({
      where: { key: 'verify_010-1234-5678' }
    });
    verifyCode = JSON.parse(setting!.value).code;
  });

  it('올바른 인증번호로 검증 성공', async () => {
    const response = await request(app)
      .post('/api/v1/auth/phone/verify')
      .send({ phone: '010-1234-5678', code: verifyCode });

    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
    expect(response.body.access_token).toBeDefined();
    expect(response.body.refresh_token).toBeDefined();
    expect(response.body.user).toBeDefined();
    expect(response.body.user.phone).toBe('010-1234-5678');
  });

  it('잘못된 인증번호로 검증 실패', async () => {
    const response = await request(app)
      .post('/api/v1/auth/phone/verify')
      .send({ phone: '010-1234-5678', code: '000000' });

    expect(response.status).toBe(400);
    expect(response.body.success).toBe(false);
    expect(response.body.error).toBe('인증번호가 일치하지 않습니다');
  });

  it('만료된 인증번호로 검증 실패', async () => {
    const setting = await prisma.setting.findUnique({
      where: { key: 'verify_010-1234-5678' }
    });
    const value = JSON.parse(setting!.value);
    value.expiresAt = new Date(Date.now() - 1000).toISOString();
    await prisma.setting.update({
      where: { key: 'verify_010-1234-5678' },
      data: { value: JSON.stringify(value) }
    });

    const response = await request(app)
      .post('/api/v1/auth/phone/verify')
      .send({ phone: '010-1234-5678', code: verifyCode });

    expect(response.status).toBe(400);
    expect(response.body.success).toBe(false);
    expect(response.body.error).toBe('인증번호가 만료되었습니다');
  });

  it('첫 인증 시 사용자 자동 생성', async () => {
    await request(app)
      .post('/api/v1/auth/phone/verify')
      .send({ phone: '010-1234-5678', code: verifyCode });

    const user = await prisma.user.findUnique({
      where: { phone: '010-1234-5678' }
    });

    expect(user).not.toBeNull();
    expect(user!.phone).toBe('010-1234-5678');
    expect(user!.isActive).toBe(true);
  });
});
```

```typescript
import request from 'supertest';
import app from '../../src/app';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

describe('POST /api/v1/orders', () => {
  let userToken: string;
  let restaurantId: string;
  let menuId: string;

  beforeEach(async () => {
    await prisma.orderItem.deleteMany();
    await prisma.order.deleteMany();
    await prisma.menu.deleteMany();
    await prisma.restaurant.deleteMany();
    await prisma.user.deleteMany();
    await prisma.setting.deleteMany();

    // 사용자 생성 및 토큰 발급
    const verifyRes = await request(app)
      .post('/api/v1/auth/phone/request')
      .send({ phone: '010-1111-2222' });
    const setting = await prisma.setting.findUnique({
      where: { key: 'verify_010-1111-2222' }
    });
    const code = JSON.parse(setting!.value).code;
    const authRes = await request(app)
      .post('/api/v1/auth/phone/verify')
      .send({ phone: '010-1111-2222', code });
    userToken = authRes.body.access_token;

    // 식당 및 메뉴 생성
    const restaurant = await prisma.restaurant.create({
      data: {
        name: '테스트 식당',
        address: '제주시 한경면',
        latitude: 33.3615,
        longitude: 126.3098,
        isActive: true,
        isDeliverable: true,
        deliveryRadius: 5.0
      }
    });
    restaurantId = restaurant.id;

    const menu = await prisma.menu.create({
      data: {
        restaurantId: restaurant.id,
        name: '테스트 메뉴',
        price: 10000,
        isAvailable: true,
        isActive: true
      }
    });
    menuId = menu.id;
  });

  it('유효한 주문 생성 성공', async () => {
    const response = await request(app)
      .post('/api/v1/orders')
      .set('Authorization', `Bearer ${userToken}`)
      .send({
        restaurantId,
        items: [{ menuId, quantity: 2 }],
        deliveryAddress: '제주시 한경면 신도리 123',
        deliveryLatitude: 33.365,
        deliveryLongitude: 126.315
      });

    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
    expect(response.body.data.orderId).toBeDefined();
    expect(response.body.data.orderNumber).toMatch(/^ORD\d{8}\d{6}$/);
    expect(response.body.data.subtotal).toBe(20000);
    expect(response.body.data.totalAmount).toBeGreaterThan(20000);
  });

  it('인증 없이 주문 시 401 에러', async () => {
    const response = await request(app)
      .post('/api/v1/orders')
      .send({
        restaurantId,
        items: [{ menuId, quantity: 1 }],
        deliveryAddress: '제주시 한경면 신도리 123',
        deliveryLatitude: 33.365,
        deliveryLongitude: 126.315
      });

    expect(response.status).toBe(401);
  });

  it('존재하지 않는 식당으로 주문 시 실패', async () => {
    const response = await request(app)
      .post('/api/v1/orders')
      .set('Authorization', `Bearer ${userToken}`)
      .send({
        restaurantId: 'invalid-id',
        items: [{ menuId, quantity: 1 }],
        deliveryAddress: '제주시 한경면 신도리 123',
        deliveryLatitude: 33.365,
        deliveryLongitude: 126.315
      });

    expect(response.status).toBe(400);
    expect(response.body.error).toContain('식당');
  });

  it('품절 메뉴 주문 시 실패', async () => {
    await prisma.menu.update({
      where: { id: menuId },
      data: { isAvailable: false }
    });

    const response = await request(app)
      .post('/api/v1/orders')
      .set('Authorization', `Bearer ${userToken}`)
      .send({
        restaurantId,

        deliveryLatitude: 33.365,

    expect(response.status).toBe(400);
    expect(response.body.error).toContain('품절');
```

```typescript
    it('같은 좌표일 때 거리 0', () => {
      const coord = { lat: 33.3615, lng: 126.3098 };
      const distance = calculateDistance(coord, coord);
      expect(distance).toBe(0);
    });

    it('1km 거리 계산 정확성', () => {
      const coord1 = { lat: 33.3615, lng: 126.3098 };
      const coord2 = { lat: 33.3705, lng: 126.3098 };
      const distance = calculateDistance(coord1, coord2);
      expect(distance).toBeCloseTo(1.0, 1);
    });

    it('5km 거리 계산', () => {
      const coord1 = { lat: 33.3615, lng: 126.3098 };
      const coord2 = { lat: 33.4065, lng: 126.3098 };
      const distance = calculateDistance(coord1, coord2);
      expect(distance).toBeCloseTo(5.0, 1);
    });
  });

  describe('calculateDeliveryFee', () => {
    it('2km 이내 기본 배달비', () => {
      const result = calculateDeliveryFee(1.5);
      expect(result.fee).toBe(3000);
      expect(result.isDeliverable).toBe(true);
    });

    it('2km 초과 시 추가 요금', () => {
      const result = calculateDeliveryFee(3.0);
      expect(result.fee).toBe(3500);
      expect(result.isDeliverable).toBe(true);
    });

    it('5km 초과 시 배달 불가', () => {
      const result = calculateDeliveryFee(5.5);
      expect(result.fee).toBe(0);
      expect(result.isDeliverable).toBe(false);
    });

    it('커스텀 기본료 적용', () => {
      const result = calculateDeliveryFee(1.0, 4000);
      expect(result.fee).toBe(4000);
    });

    it('커스텀 km당 요금 적용', () => {
      const result = calculateDeliveryFee(3.0, 3000, 1000);
      expect(result.fee).toBe(4000);
    });
  });
```

```typescript
// tests/auth/jwt.test.ts
import { JWTTokenService } from '../../src/services/JWTTokenService';

describe('JWT Token Service', () => {
  const jwtService = new JWTTokenService();

  describe('Access Token', () => {
    it('토큰 생성 성공', () => {
      const token = jwtService.generateAccessToken('user-123', '010-1234-5678');
      expect(token).toBeDefined();
      expect(typeof token).toBe('string');
    });

    it('유효한 토큰 검증 성공', () => {
      const token = jwtService.generateAccessToken('user-123', '010-1234-5678');
      const decoded = jwtService.verifyToken(token);
      expect(decoded.userId).toBe('user-123');
      expect(decoded.phone).toBe('010-1234-5678');
    });

    it('잘못된 토큰 검증 실패', () => {
      expect(() => {
        jwtService.verifyToken('invalid-token');
      }).toThrow();
    });
  });

  describe('Refresh Token', () => {
    it('리프레시 토큰으로 액세스 토큰 갱신', () => {
      const refreshToken = jwtService.generateRefreshToken('user-123', '010-1234-5678');
      const newAccessToken = jwtService.refreshAccessToken(refreshToken);
      expect(newAccessToken).toBeDefined();
    });
  });

  describe('Confirm Token', () => {
    it('주문 확인 토큰 생성', () => {
      const token = jwtService.generateConfirmToken('order-123', '010-1234-5678');
      expect(token).toBeDefined();
    });

    it('주문 확인 토큰 검증', () => {
      const token = jwtService.generateConfirmToken('order-123', '010-1234-5678');
      const decoded = jwtService.verifyConfirmToken(token);
      expect(decoded.orderId).toBe('order-123');
    });
  });
});
```
